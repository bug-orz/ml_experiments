def Gini(dataset,feature,value):
    stack=[]
    stackDic={}
    for row in dataset.itertuples():
        if value in row:
            stack.append(row[1:])
    for temp in stack:
        if(temp not in stackDic.keys()):
            stackDic[temp]=1
        else:
            stackDic[temp]+=1
    p=0
    for temp in stackDic.keys():
        p+=(stackDic[temp]*stackDic[temp]/(len(stackDic)*len(stackDic)))
    return round(1-p,2)
